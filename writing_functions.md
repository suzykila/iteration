z_scores=function(x){
  
  if (!is.numeric(x)){
    stop("x needs to be numeric")
  }
  
  if (length(x)<5){
    stop("you need at least 5 numbers to compute z scores")
  }
  
  z= (x-mean(x))/sd(x)
  
  return(z)
}
